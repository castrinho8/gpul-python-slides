Como return pero para xeradores.

def f_iterative(n):
	if n < 0:
		raise ValueError("f is only defined for positive numbers.")
	val = 1
	while n > 1:
		val *= n
		n = n - 1
	return val 
def f_explain(n):
	val = 1
	while n > 1:
		val *= n
		n = n - 1
		yield val
	yield val

NO RETURN.
SyntaxError: 'return' with argument inside generator

Esta traspa ten 3 pasos.
  • Non se pode usar 'return'
  • Múltiples "returns"
  • Returns "parciais"
  • yield recorda onde estaba
  • O obxecto devolto por f_explain é un obxecto generator.

Exemplo real

class Node(object):
	def __init__(self, left=None, right=None):
		self.left = left, self.right = right
		# Anything else...

	def get_inorder_list(self):
		'''
		Returns a list of the order in
		in-order position
		'''
		for elt in this.left.get_inorder_list():
			# Yield all elements before me, inorder
			yield elt

		yield self # Yield ourselves

		for elt in this.right.get_inorder_list():
			# Yield all elements after me, inorder
			yield elt

Módulo unittest

import random
import unittest

class TestSequenceFunctions(unittest.TestCase):

  def setUp(self):
    self.seq = range(10)

  def test_shuffle(self):
    random.shuffle(self.seq)
    self.seq.sort()
    self.assertEqual(self.seq, range(10))

    self.assertRaises(TypeError, random.shuffle, (1,2,3))
